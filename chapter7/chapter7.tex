\[
\begin{aligned}
\begin{split}
&D(s)=-2-s+2s^2+s^3\\

&\overline{D}(s)=\overline{D}_0+\overline{D}_1s+\overline{D}_2s^2\\
&\overline{N}(s)=\overline{N}_0+\overline{N}_1s+\overline{N}_2s^2\\
\end{split}
\end{aligned}
\]

\[
\textbf{SM}:=
\left[
